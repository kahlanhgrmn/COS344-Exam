1. The OpenGL State Machine Concept

OpenGL is a state machine: every setting you change persists until you change it again. The current state determines how every draw call behaves.

Key State Variables

Exam tip: State leaks between draw calls. If you enable blending for one object and forget to disable it, the next object will also blend.

2. Object Creation & Binding Pattern

VAO (Vertex Array Object)

Stores the configuration of vertex attribute pointers and which VBO they point to.


GLuint vao;
glGenVertexArrays(1, &vao);
glBindVertexArray(vao);       // Bind FIRST before configuring VBO

VBO (Vertex Buffer Object)

Stores raw vertex data on the GPU.


GLuint vbo;
glGenBuffers(1, &vbo);
glBindBuffer(GL_ARRAY_BUFFER, vbo);
glBufferData(GL_ARRAY_BUFFER,       // target
             sizeof(vertices),      // size in BYTES
             vertices,              // pointer to CPU data
             GL_STATIC_DRAW);       // usage hint

GL_STATIC_DRAW = data set once, used many times. Also: GL_DYNAMIC_DRAW, GL_STREAM_DRAW.

EBO (Element Buffer Object) — for indexed drawing


GLuint ebo;
glGenBuffers(1, &ebo);
glBindBuffer(GL_ELEMENT_ARRAY_BUFFER, ebo);
glBufferData(GL_ELEMENT_ARRAY_BUFFER, sizeof(indices), indices, GL_STATIC_DRAW);

Vertex Attribute Pointer

Tells OpenGL how to interpret raw bytes as attributes:


glVertexAttribPointer(
    0,                  // location (matches layout(location=0) in shader)
    3,                  // number of components (vec3 = 3 floats)
    GL_FLOAT,           // data type
    GL_FALSE,           // normalize?
    3 * sizeof(float),  // stride: bytes between consecutive vertices
    (void*)0            // offset: bytes from start to this attribute
);
glEnableVertexAttribArray(0);

Interleaved attributes (position + normal + texcoord in one VBO):


// Layout: [x,y,z, nx,ny,nz, u,v] per vertex  =  8 floats = 32 bytes stride

// Position at location 0
glVertexAttribPointer(0, 3, GL_FLOAT, GL_FALSE, 8*sizeof(float), (void*)0);
glEnableVertexAttribArray(0);

// Normal at location 1
glVertexAttribPointer(1, 3, GL_FLOAT, GL_FALSE, 8*sizeof(float), (void*)(3*sizeof(float)));
glEnableVertexAttribArray(1);

// TexCoord at location 2
glVertexAttribPointer(2, 2, GL_FLOAT, GL_FALSE, 8*sizeof(float), (void*)(6*sizeof(float)));
glEnableVertexAttribArray(2);

Exam trap: Always bind the VAO before configuring VBO attributes. The VAO records which VBO is bound and how attributes are laid out.

3. Shader Compilation Pipeline

Step-by-step (9 steps)


// 1. Create shader object
GLuint vertShader = glCreateShader(GL_VERTEX_SHADER);

// 2. Attach source code
glShaderSource(vertShader, 1, &vertSrc, NULL);

// 3. Compile
glCompileShader(vertShader);

// 4. Check for errors
GLint success;
glGetShaderiv(vertShader, GL_COMPILE_STATUS, &success);
if (!success) {
    char log[512];
    glGetShaderInfoLog(vertShader, 512, NULL, log);
    std::cout << "SHADER ERROR: " << log << std::endl;
}

// 5. Repeat for fragment shader
GLuint fragShader = glCreateShader(GL_FRAGMENT_SHADER);
glShaderSource(fragShader, 1, &fragSrc, NULL);
glCompileShader(fragShader);

// 6. Create program
GLuint shaderProg = glCreateProgram();

// 7. Attach both shaders
glAttachShader(shaderProg, vertShader);
glAttachShader(shaderProg, fragShader);

// 8. Link
glLinkProgram(shaderProg);

// 9. Use
glUseProgram(shaderProg);

// Cleanup (optional but good practice)
glDeleteShader(vertShader);
glDeleteShader(fragShader);

Minimal Vertex Shader


#version 330 core
layout(location = 0) in vec3 aPos;

uniform mat4 model;
uniform mat4 view;
uniform mat4 projection;

void main() {
    gl_Position = projection * view * model * vec4(aPos, 1.0);
}

Minimal Fragment Shader


#version 330 core
out vec4 FragColor;

void main() {
    FragColor = vec4(1.0, 0.5, 0.2, 1.0);  // orange
}

    │               Transforms to clip space (gl_Position)
    │               Passes interpolatable data (normals, UVs)
    ▼

    │
    ▼

    │

5. Uniforms and Attribute Locations

Setting Uniforms


// Must call glUseProgram FIRST
glUseProgram(shaderProg);

GLint loc = glGetUniformLocation(shaderProg, "model");
SpaceObtained ByRange

Exam trap: glUniformMatrix4fv — the 3rd arg is GL_FALSE (OpenGL uses column-major, which matches GLM's layout). GL_TRUE would transpose.

Attribute Locations

Two ways to specify which location maps to which shader input:

Method 1: Layout qualifier in GLSL (preferred)


layout(location = 0) in vec3 aPos;
layout(location = 1) in vec3 aNormal;

Method 2: Query from CPU


GLint posLoc = glGetAttribLocation(shaderProg, "aPos");
glVertexAttribPointer(posLoc, 3, GL_FLOAT, GL_FALSE, ...);

6. Texture Setup

Complete Texture Setup


// 1. Generate
GLuint texture;
glGenTextures(1, &texture);

// 2. Bind
glBindTexture(GL_TEXTURE_2D, texture);

// 3. Wrapping parameters
glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_WRAP_S, GL_REPEAT);
glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_WRAP_T, GL_REPEAT);

// 4. Filtering parameters
glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_MIN_FILTER, GL_LINEAR_MIPMAP_LINEAR);
glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_MAG_FILTER, GL_LINEAR);

// 5. Load image data (e.g., using stb_image)
int width, height, nChannels;
unsigned char* data = stbi_load("texture.png", &width, &height, &nChannels, 0);
glTexImage2D(GL_TEXTURE_2D, 0, GL_RGB, width, height, 0, GL_RGB, GL_UNSIGNED_BYTE, data);

// 6. Generate mipmaps
glGenerateMipmap(GL_TEXTURE_2D);
stbi_image_free(data);

Using Textures in Render Loop


// Activate texture unit, then bind
glActiveTexture(GL_TEXTURE0);          // Select unit 0
glBindTexture(GL_TEXTURE_2D, texture); // Bind texture to unit 0

// Tell shader which unit to sample from
glUniform1i(glGetUniformLocation(shader, "texture1"), 0);  // unit 0

7. Framebuffer & Depth Test


// Enable depth testing (do once at init)
glEnable(GL_DEPTH_TEST);
glDepthFunc(GL_LESS);   // Pass if fragment z < z-buffer z

// At start of each frame
glClearColor(0.1f, 0.1f, 0.1f, 1.0f);
glClear(GL_COLOR_BUFFER_BIT | GL_DEPTH_BUFFER_BIT);

glDepthFunc Options

TypePurposeHow to Apply
Diffuse/Albedo mapBase colorMultiply by lighting result
Normal mapPer-pixel normalsSample, remap [0,1]→[-1,1], transform to world space
Bump mapHeight-based normals (older)Derive normal from height differences
Displacement mapActually moves geometryApplied in vertex/tessellation shader

Exam tip: glClearColor sets what color to fill with. glClear executes the fill. Both are needed.

8. Backface Culling


glEnable(GL_CULL_FACE);
glCullFace(GL_BACK);     // Cull back-facing triangles (default)
glFrontFace(GL_CCW);     // Counter-clockwise winding = front face (default)

A triangle's winding order (when viewed from outside) determines front vs back:

  • Counter-clockwise (CCW) → front face → rendered
  • Clockwise (CW) → back face → culled

Exam tip: Backface culling halves the fragment shader load. It exploits the property that for closed meshes, back faces are never visible.

9. Render Loop Structure


while (!glfwWindowShouldClose(window)) {
    // 1. Process input
    processInput(window);

    // 2. Clear
    glClearColor(0.1f, 0.1f, 0.1f, 1.0f);
    glClear(GL_COLOR_BUFFER_BIT | GL_DEPTH_BUFFER_BIT);

    // 3. Activate shader program
    glUseProgram(shaderProg);

    // 4. Set uniforms
    glm::mat4 model = glm::rotate(glm::mat4(1.0f), (float)glfwGetTime(), glm::vec3(0,1,0));
    glUniformMatrix4fv(glGetUniformLocation(shaderProg, "model"), 1, GL_FALSE, glm::value_ptr(model));
    // ... view and projection similarly

    // 5. Bind VAO (restores vertex attribute config)
    glBindVertexArray(vao);

    // 6. Draw
    glDrawArrays(GL_TRIANGLES, 0, 36);
    // OR: glDrawElements(GL_TRIANGLES, indexCount, GL_UNSIGNED_INT, 0);

    // 7. Swap buffers (shows back buffer on screen)
    glfwSwapBuffers(window);

    // 8. Poll events (keyboard, mouse, window resize)
    glfwPollEvents();
}

11. Viewport Transform

Maps NDC [-1,1]² to screen pixels:


glViewport(x, y, width, height);

Matrix form:


Mvp = | w/2   0   0   x+w/2 |
      |  0   h/2  0   y+h/2 |
      |  0    0  1/2   1/2  |
      |  0    0   0     1   |

Exam tip: The viewport transform is the last step after the perspective divide. It's fixed-function — you don't write a shader for it.

13. Complete OpenGL Initialization Sequence


1.  glfwInit()
2.  glfwCreateWindow()
3.  glfwMakeContextCurrent()
4.  gladLoadGLLoader() or glewInit()
5.  glViewport()
6.  glEnable(GL_DEPTH_TEST)
7.  Create + compile vertex shader
8.  Create + compile fragment shader
9.  Create + link shader program
10. glGenVertexArrays() + glBindVertexArray()
11. glGenBuffers() + glBindBuffer() + glBufferData()  [VBO]
12. glGenBuffers() + glBindBuffer() + glBufferData()  [EBO, if needed]
13. glVertexAttribPointer() + glEnableVertexAttribArray()  [per attribute]
14. Load textures: glGenTextures → glBindTexture → glTexImage2D → glGenerateMipmap
15. Enter render loop
    a. glClear()
    b. glUseProgram()
    c. glUniform*() for all uniforms
    d. glBindVertexArray()
    e. glDrawArrays() / glDrawElements()
    f. glfwSwapBuffers()
    g. glfwPollEvents()
16. glDeleteVertexArrays / glDeleteBuffers / glDeleteProgram
17. glfwTerminate()
